package com.method;

public class Employee {

	int myCube(int num) {

		int cube = num * num * num;

		return cube;

	}

	int myfact(int num) {
		int fact = 1;
		for (int i = 1; i <= num; i++) {
			fact = fact * i;
		}

		return fact;

	}

	String myName() {
		return "Om Patil";
	}

	String myName(String fName, String lName) {
		return fName + " " + lName;
	}

	void myInfo(String name, int myAge, long mobNo, String city, String email, char gender) {
		System.out.println("My Name is " + name);
		System.out.println("My Age is " + myAge);
		System.out.println("My Mobile Num is " + mobNo);
		System.out.println("My city is " + city);
		System.out.println("My email is " + email);
		System.out.println("My Gender is " + gender);
	}

}
